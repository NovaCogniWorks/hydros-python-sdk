## 🔄 生命周期方法

### 初始化

```python
@handle_agent_errors(ErrorCodes.AGENT_INIT_FAILURE)
def on_init(self, request: SimTaskInitRequest) -> SimTaskInitResponse:
    # 1. 加载配置
    self.load_agent_configuration(request)

    # 2. 加载拓扑（可选）
    topology_url = self.properties.get_property('hydros_objects_modeling_url')
    if topology_url:
        self._topology = HydroObjectUtilsV2.build_waterway_topology(topology_url)

    # 3. 初始化模型
    self._initialize_model()

    # 4. 注册状态管理器
    self.state_manager.init_task(self.context, [self])
    self.state_manager.add_local_agent(self)

    # 5. 返回响应
    return SimTaskInitResponse(
        context=self.context,
        command_id=request.command_id,
        command_status=CommandStatus.SUCCEED,
        source_agent_instance=self,
        broadcast=False
    )
```

### 执行仿真

```python
@handle_agent_errors(ErrorCodes.AGENT_TICK_FAILURE)
def on_tick_simulation(self, request: TickCmdRequest) -> List[MqttMetrics]:
    step = request.step

    # 1. 收集边界条件
    boundary_conditions = self._collect_boundary_conditions(step)

    # 2. 执行仿真
    results = self._execute_simulation(step, boundary_conditions)

    # 3. 转换为指标
    metrics_list = self._convert_results_to_metrics(results, step)

    return metrics_list
```

### 终止清理

```python
@handle_agent_errors(ErrorCodes.AGENT_TERMINATE_FAILURE)
def on_terminate(self, request: SimTaskTerminateRequest) -> SimTaskTerminateResponse:
    # 1. 清理资源
    if self._my_model:
        self._my_model.cleanup()

    # 2. 注销状态管理器
    self.state_manager.terminate_task(self.context)
    self.state_manager.remove_local_agent(self)

    # 3. 返回响应
    return SimTaskTerminateResponse(
        context=self.context,
        command_id=request.command_id,
        command_status=CommandStatus.SUCCEED,
        source_agent_instance=self,
        broadcast=False
    )
```

## 📈 边界条件处理

### 收集边界条件

```python
def _collect_boundary_conditions(self, step: int) -> Dict[int, Dict[str, float]]:
    boundary_conditions = {}

    # 定义需要的指标
    bc_metrics = ['inflow', 'water_level', 'gate_opening']

    # 遍历拓扑对象
    if self._topology:
        for top_obj in self._topology.top_objects:
            for child in top_obj.children:
                object_bc = {}

                for metrics_code in bc_metrics:
                    # 从缓存获取值
                    value = self.get_time_series_value(
                        child.object_id,
                        metrics_code,
                        step
                    )

                    if value is not None:
                        object_bc[metrics_code] = value

                if object_bc:
                    boundary_conditions[child.object_id] = object_bc

    return boundary_conditions
```

def on_boundary_condition_update(self, time_series_list: List[ObjectTimeSeries]):
    for time_series in time_series_list:
        logger.info(
            f"更新: object={time_series.object_name}, "
            f"metrics={time_series.metrics_code}, "
            f"values={len(time_series.time_series)}"
        )

## 🚀 启动脚本模板

```python
import os
import time
import logging
from hydros_agent_sdk import (
    setup_logging,
    SimCoordinationClient,
    HydroAgentFactory,
    MultiAgentCallback,
    load_env_config,
)
from my_agent import MyAgent

# 配置日志
setup_logging(level=logging.INFO, console=True)
logger = logging.getLogger(__name__)

def main():
    # 加载配置
    script_dir = os.path.dirname(os.path.abspath(__file__))
    env_config = load_env_config(os.path.join(script_dir, "env.properties"))

    # 创建工厂
    agent_factory = HydroAgentFactory(
        agent_class=MyAgent,
        config_file=os.path.join(script_dir, "agent.properties"),
        env_config=env_config
    )

    # 创建回调
    callback = MultiAgentCallback(node_id="LOCAL")
    callback.register_agent_factory("MY_AGENT", agent_factory)

    # 创建客户端
    client = SimCoordinationClient(
        broker_url=env_config['mqtt_broker_url'],
        broker_port=int(env_config['mqtt_broker_port']),
        topic=env_config.get('mqtt_topic', f"/hydros/commands/coordination/{env_config['hydros_cluster_id']}"),
        sim_coordination_callback=callback
    )

    callback.set_client(client)

    # 启动服务
    try:
        logger.info("启动服务...")
        client.start()
        logger.info("服务已启动，按 Ctrl+C 停止")

        while True:
            time.sleep(1)

    except KeyboardInterrupt:
        logger.info("停止服务...")
        client.stop()

if __name__ == "__main__":
    main()
```
